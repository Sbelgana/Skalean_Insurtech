## Objectif Global du Sprint

Construire l'application **Assurflow Tow Mobile PWA** (port 3007) Next.js 15 mobile-first Uber-style : interface dediee aux remorqueurs (acteur 5/6 ecosystem Assurflow v3.0) pour reception missions push, geolocation real-time, navigation GPS, photos before/after, rating bidirectionnel. Application installable sur mobile (PWA) avec biometric auth (WebAuthn).

A la sortie de ce sprint :
- App `web-tow-mobile` Next.js 15 PWA installable (port 3007 dev / `app.assurflow.com/tow-mobile` prod)
- 12 pages applicatives : login + MFA + dashboard driver + mission active + missions history + earnings + profile + KYB + ratings + map + navigation + parametres
- 3 roles UI : tow_admin (gestionnaire flotte) / tow_driver (chauffeur PWA) / tow_dispatcher (coordination)
- Backend complete : 4 entites DB (tow_drivers + tow_missions + tow_vehicle_proofs + tow_ratings)
- Workflow Uber-style : dispatch automatique driver le plus proche selon geoloc + rating + availability
- Cross-tenant : client_to_tower_dispatch (mission) + tower_to_garage_delivery (livraison)
- Tarification dynamic : prix calcule sur distance + heure (jour/nuit) + urgence (standard/urgent)
- Photos before/after obligatoires avec timestamp + geoloc proof
- Paiement integration Sprint 11 (CMI/YouCan/Mobile Money) -- garage paye remorqueur ou client direct
- Rating bidirectionnel : client note driver + driver note customer
- Push notifications PWA : nouvelle mission + acceptation client + arrivee
- Offline-first : driver peut accepter mission sans connexion (sync auto retour online)
- I18n fr/ar-MA/ar (driver lit darija)
- Tests Playwright E2E + WCAG 2.1 AA + Lighthouse PWA >= 95
- WebAuthn biometric pour driver (face/fingerprint)

## Differentiation strategique

**Aucun concurrent local au Maroc** n'a app remorqueur dediee. Courtizen n'aborde pas ce sujet. Les remorqueurs MA travaillent actuellement :
- Appels telephone manuels
- Pas de tracking
- Pas de photos before/after (litiges frequents)
- Tarification opaque (negociation chaque fois)
- Pas de rating
- Paiement cash uniquement (pas de trace fiscale)

**Assurflow Tow resout 6 pain points** :
1. Dispatch automatique driver le plus proche -- delai 30 min vs 2-3h actuel
2. Tracking GPS real-time -- client sait toujours ou est sa voiture
3. Photos before/after timestamp + geoloc = preuves anti-litige
4. Tarification transparente (algorithme distance + heure + urgence)
5. Rating bidirectionnel = qualite service garantie
6. Paiement traceable + factures auto = conformite fiscale

**Effet marketing** : "Uber des remorqueurs au Maroc" = positionnement instantanement comprehensible.

-- tow_missions : Missions de remorquage
CREATE TABLE tow_missions (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id uuid NOT NULL REFERENCES auth_tenants(id) ON DELETE CASCADE,
  
  -- Acteurs
  customer_tenant_id uuid NOT NULL REFERENCES auth_tenants(id),
  customer_user_id uuid NOT NULL REFERENCES auth_users(id),
  driver_id uuid REFERENCES tow_drivers(id) ON DELETE SET NULL,
  destination_garage_tenant_id uuid REFERENCES auth_tenants(id),
  
  -- Sinistre lie (optionnel)
  sinistre_id uuid,  -- FK insure_sinistres apres Sprint 14
  
  -- Localisations
  pickup_lat decimal(10, 7) NOT NULL,
  pickup_lng decimal(10, 7) NOT NULL,
  pickup_address text NOT NULL,
  destination_lat decimal(10, 7) NOT NULL,
  destination_lng decimal(10, 7) NOT NULL,
  destination_address text NOT NULL,
  distance_km decimal(8, 3),  -- calcule par DispatchService
  
  -- Vehicule a remorquer
  vehicle_immatriculation text NOT NULL,
  vehicle_brand text,
  vehicle_model text,
  vehicle_color text,
  vehicle_problem_description text,
  
  -- Workflow status
  status text NOT NULL CHECK (status IN (
    'pending',           -- mission creee, en attente dispatch
    'dispatched',        -- driver designe, attente acceptation
    'accepted',          -- driver accepte
    'rejected',          -- driver rejete
    'en_route_to_pickup', -- driver va chercher vehicule
    'at_pickup',         -- driver arrive lieu pickup
    'loading',           -- driver charge vehicule
    'en_route_to_destination', -- driver achemine
    'at_destination',    -- driver arrive lieu destination
    'unloading',         -- driver decharge
    'completed',         -- mission terminee
    'cancelled_by_customer',
    'cancelled_by_driver',
    'cancelled_by_admin',
    'no_driver_available' -- aucun driver dispo apres 3 tentatives
  )) DEFAULT 'pending',
  
  -- Timestamps workflow
  created_at timestamptz NOT NULL DEFAULT NOW(),
  dispatched_at timestamptz,
  accepted_at timestamptz,
  pickup_reached_at timestamptz,
  loading_completed_at timestamptz,
  destination_reached_at timestamptz,
  completed_at timestamptz,
  cancelled_at timestamptz,
  cancellation_reason text,
  
  -- Tarification
  price_base_mad decimal(10, 2) NOT NULL,
  price_distance_mad decimal(10, 2) NOT NULL,
  price_time_modifier_mad decimal(10, 2) NOT NULL DEFAULT 0,  -- nuit/weekend
  price_urgency_modifier_mad decimal(10, 2) NOT NULL DEFAULT 0,
  price_total_mad decimal(10, 2) NOT NULL,
  payer text NOT NULL CHECK (payer IN ('customer', 'garage', 'carrier')),
  payment_transaction_id uuid,  -- FK pay_transactions Sprint 11
  payment_status text NOT NULL CHECK (payment_status IN ('pending', 'completed', 'refunded', 'failed')) DEFAULT 'pending',
  
  -- Urgence + type
  urgency text NOT NULL CHECK (urgency IN ('standard', 'urgent')) DEFAULT 'standard',
  
  -- Metadata
  notes_customer text,
  notes_driver text,
  updated_at timestamptz NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_tow_missions_tenant ON tow_missions(tenant_id);
CREATE INDEX idx_tow_missions_customer ON tow_missions(customer_tenant_id, customer_user_id);
CREATE INDEX idx_tow_missions_driver ON tow_missions(driver_id);
CREATE INDEX idx_tow_missions_status ON tow_missions(status);
CREATE INDEX idx_tow_missions_destination_garage ON tow_missions(destination_garage_tenant_id);

ALTER TABLE tow_missions ENABLE ROW LEVEL SECURITY;

-- tow_vehicle_proofs : Photos before/after avec EXIF + geoloc
CREATE TABLE tow_vehicle_proofs (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id uuid NOT NULL REFERENCES auth_tenants(id) ON DELETE CASCADE,
  mission_id uuid NOT NULL REFERENCES tow_missions(id) ON DELETE CASCADE,
  
  proof_type text NOT NULL CHECK (proof_type IN ('before_pickup', 'after_loading', 'during_transport', 'before_unloading', 'after_delivery')),
  
  -- Photo URL + metadata
  photo_url text NOT NULL,
  photo_size_bytes integer,
  photo_mime_type text NOT NULL,
  
  -- EXIF / geoloc proof (anti-litige)
  exif_timestamp timestamptz NOT NULL,  -- timestamp pris au moment photo (PAS upload)
  exif_lat decimal(10, 7) NOT NULL,
  exif_lng decimal(10, 7) NOT NULL,
  exif_device_model text,
  exif_orientation integer,
  
  -- Validation server-side
  geoloc_match_pickup boolean,  -- coords EXIF coincident avec pickup_lat/lng
  geoloc_match_destination boolean,
  timestamp_in_mission_window boolean,  -- timestamp EXIF entre dispatched_at et completed_at
  
  -- Upload metadata
  uploaded_by_driver_id uuid NOT NULL REFERENCES tow_drivers(id),
  uploaded_at timestamptz NOT NULL DEFAULT NOW(),
  
  notes text

-- tow_ratings : Rating bidirectionnel (customer rate driver + driver rate customer)
CREATE TABLE tow_ratings (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id uuid NOT NULL REFERENCES auth_tenants(id) ON DELETE CASCADE,
  mission_id uuid NOT NULL REFERENCES tow_missions(id) ON DELETE CASCADE,
  
  rated_by text NOT NULL CHECK (rated_by IN ('customer', 'driver')),
  rater_user_id uuid NOT NULL REFERENCES auth_users(id),
  rated_user_id uuid NOT NULL REFERENCES auth_users(id),
  
  rating_score integer NOT NULL CHECK (rating_score >= 1 AND rating_score <= 5),
  rating_comment text,
  
  -- Categories specifiques (customer rates driver)
  driver_punctuality integer CHECK (driver_punctuality >= 1 AND driver_punctuality <= 5),
  driver_professionalism integer CHECK (driver_professionalism >= 1 AND driver_professionalism <= 5),
  driver_vehicle_care integer CHECK (driver_vehicle_care >= 1 AND driver_vehicle_care <= 5),
  driver_communication integer CHECK (driver_communication >= 1 AND driver_communication <= 5),
  
  -- Categories specifiques (driver rates customer)
  customer_punctuality integer CHECK (customer_punctuality >= 1 AND customer_punctuality <= 5),
  customer_communication integer CHECK (customer_communication >= 1 AND customer_communication <= 5),
  customer_payment_behavior integer CHECK (customer_payment_behavior >= 1 AND customer_payment_behavior <= 5),
  
  created_at timestamptz NOT NULL DEFAULT NOW(),
  UNIQUE(mission_id, rated_by)  -- 1 rating max par cote
);

CREATE INDEX idx_tow_ratings_mission ON tow_ratings(mission_id);
CREATE INDEX idx_tow_ratings_rated_user ON tow_ratings(rated_user_id);

ALTER TABLE tow_ratings ENABLE ROW LEVEL SECURITY;
ALTER TABLE tow_ratings FORCE ROW LEVEL SECURITY;

### Pattern critique : Algorithme dispatch

```typescript
async dispatchMission(missionId: string): Promise<DispatchResult> {
  const mission = await this.missionsRepo.findOneOrFail(missionId);
  
  // Step 1 : find available drivers in 10km radius
  let drivers = await this.findAvailableDriversInRadius(
    mission.pickup_lat,
    mission.pickup_lng,
    10
  );
  
  if (drivers.length === 0) {
    // Expand radius progressively
    drivers = await this.findAvailableDriversInRadius(mission.pickup_lat, mission.pickup_lng, 20);
  }
  
  if (drivers.length === 0) {
    await this.markMissionAsNoDriverAvailable(missionId);
    return { success: false, reason: 'no_driver_available' };
  }
  
  // Step 2 : sort drivers
  drivers.sort((a, b) => {
    // Primary : distance ASC
    const distDiff = a.distanceKm - b.distanceKm;
    if (Math.abs(distDiff) > 0.5) return distDiff;
    
    // Secondary : rating DESC
    const ratingDiff = (b.avg_rating ?? 3) - (a.avg_rating ?? 3);
    if (Math.abs(ratingDiff) > 0.1) return ratingDiff;
    
    // Tertiary : last_mission_completed_at DESC (most recent first)
    return new Date(b.last_mission_completed_at ?? 0).getTime() 
         - new Date(a.last_mission_completed_at ?? 0).getTime();
  });
  
  // Step 3 : try top 3 drivers with 60s timeout each
  for (let i = 0; i < Math.min(3, drivers.length); i++) {
    const driver = drivers[i];
    
    // Update mission status
    await this.missionsRepo.update(missionId, { 
      driver_id: driver.id, 
      status: 'dispatched', 
      dispatched_at: new Date() 
    });
    
    // Notify via Web Push
    await this.notificationsService.notifyDriverNewMission(driver.id, missionId);
    
    // Wait for acceptance (poll mission status every 5s, timeout 60s)
    const accepted = await this.waitForAcceptance(missionId, 60_000);
    
    if (accepted) {
      // Emit event
      await this.kafkaProducer.emit('insurtech.events.tow.mission.accepted', { 
        missionId, driverId: driver.id 
      });
      return { success: true, driverId: driver.id };
    }
    
    // Driver did not accept in time -> try next
    await this.missionsRepo.update(missionId, { driver_id: null, status: 'pending' });
  }
  
  // Step 4 : exhausted 3 tries
  await this.markMissionAsNoDriverAvailable(missionId);
  return { success: false, reason: 'no_driver_available' };
}
```

### Pattern critique : PricingBreakdown

```typescript
import Decimal from 'decimal.js';
import { formatInTimeZone, isWeekend, getHours } from 'date-fns-tz';

interface PricingBreakdown {
  baseMad: Decimal;
  distanceMad: Decimal;
  timeModifierMad: Decimal;
  urgencyModifierMad: Decimal;
  totalMad: Decimal;
}

computePrice(input: {
  distanceKm: number;
  pickupTime: Date;
  urgency: 'standard' | 'urgent';
}): PricingBreakdown {
  const baseFee = new Decimal(this.config.basefee ?? 50);  // 50 MAD
  const ratePerKm = new Decimal(this.config.ratePerKm ?? 8);  // 8 MAD/km
  const distance = new Decimal(input.distanceKm);
  
  // Base + distance
  const baseMad = baseFee;
  const distanceMad = ratePerKm.times(distance);
  
  // Time modifier
  const inCasablanca = formatInTimeZone(input.pickupTime, 'Africa/Casablanca', 'HH');
  const hour = parseInt(inCasablanca, 10);
  const isNight = hour >= 22 || hour < 6;
  const isWeekendDay = isWeekend(input.pickupTime);
  
  let timeModifierPct = new Decimal(0);
  if (isNight) timeModifierPct = timeModifierPct.plus(new Decimal(this.config.nightModifierPct ?? 30));
  if (isWeekendDay) timeModifierPct = timeModifierPct.plus(new Decimal(this.config.weekendModifierPct ?? 20));
  
  const subtotalBeforeTimeModifier = baseMad.plus(distanceMad);
  const timeModifierMad = subtotalBeforeTimeModifier.times(timeModifierPct).dividedBy(100);
  
  // Urgency modifier
  const urgencyModifierPct = input.urgency === 'urgent' 
    ? new Decimal(this.config.urgencyUrgentModifierPct ?? 50)
    : new Decimal(0);
  const urgencyModifierMad = subtotalBeforeTimeModifier.times(urgencyModifierPct).dividedBy(100);
  
  // Total
  const totalMad = baseMad.plus(distanceMad).plus(timeModifierMad).plus(urgencyModifierMad);
  
  return {
    baseMad: baseMad.toDecimalPlaces(2),
    distanceMad: distanceMad.toDecimalPlaces(2),
    timeModifierMad: timeModifierMad.toDecimalPlaces(2),
    urgencyModifierMad: urgencyModifierMad.toDecimalPlaces(2),
    totalMad: totalMad.toDecimalPlaces(2),
  };
}
```
