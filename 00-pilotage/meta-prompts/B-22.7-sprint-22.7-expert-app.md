## Objectif Global du Sprint

Construire l'application **Assurflow Expert** (desktop + mobile PWA) pour les experts auto agrees ACAPS et experts internes carriers. C'est l'acteur **CENTRAL** du workflow sinistre v3.0 (decision-013) : c'est l'expert qui valide/modifie/rejete le devis garage, pas le carrier directement.

A la sortie de ce sprint :
- App `web-expert` desktop Next.js 15 (port 3009 dev / `app.assurflow.com/expert` prod)
- App `web-expert-mobile` PWA mobile-first (port 3010 dev / `app.assurflow.com/expert-mobile` prod)
- 10 pages applicatives : login + MFA + dashboard missions + mission active + rapport expertise + validation devis + signature electronique + history + earnings + parametres
- 4 roles UI : expert_independent (solo) / expert_firm_admin (gerant cabinet) / expert_associate (associate) / expert_carrier_internal (salarie carrier)
- Backend complete : 4 entites DB (insure_experts + insure_expert_assignments + insure_expert_reports + insure_expert_signatures)
- Workflow v3.0 : carrier designe expert -> expert accepte/refuse -> visite garage -> rapport digital -> validation devis -> signature Barid eSign
- Cross-tenant : garage_to_expert_request (garage communique avec expert) + carrier_to_expert_designation (existant via insure_expert_assignments)
- Signature electronique Barid eSign loi 43-20 sur rapports
- Notifications multi-canal (Web Push + email + SMS)
- I18n fr/ar-MA/ar
- Tests E2E + WCAG + Lighthouse

## Differentiation strategique

**Realite terrain Maroc** (29 corrections Saad) : tout sinistre auto MA passe par expert agree ACAPS designe par le carrier. C'est LA verite que les concurrents ignorent.

**Pain points experts actuels** :
1. Designations par fax/email/whatsapp manuels
2. Photos sinistres recues par WhatsApp -- pas tracables, qualite variable
3. Rapport expertise sur Word/Excel -- pas standardise
4. Signature scannee non legale -- litiges frequents
5. Honoraires facturation manuelle papier
6. Suivi dossiers via spreadsheet personnel
7. Pas de visibility sur la queue de missions

**Assurflow Expert resout 7 pain points** :
1. Designation automatique via Carrier Portal (Sprint 26.5) avec notification push
2. Photos garage uploaded directement dans app -- HD + EXIF + tracability
3. Rapport expertise standardise digital avec photos integrees
4. Signature electronique Barid eSign loi 43-20 = valeur juridique probante
5. Honoraires factures generees automatiquement vers carriers
6. Dashboard missions queue + historique + analytics
7. Calendar integration + Maps + navigation vers garages

**Effet marketing** : "Premiere plateforme InsurTech MA respectant le workflow expert" = pitch carriers + ACAPS immediat.

  acaps_agrement_document_url text NOT NULL,
  acaps_agrement_expiry_date date NOT NULL,
  acaps_specialty text[] NOT NULL,  -- 'auto', 'incendie', 'multirisque', etc.

  firm_name text,
  firm_address text,
  firm_phone text,
  firm_email text,
  
  -- Type
  expert_type text NOT NULL CHECK (expert_type IN ('independent', 'firm_admin', 'associate', 'carrier_internal')),
  carrier_tenant_id uuid REFERENCES auth_tenants(id),  -- NOT NULL si expert_type='carrier_internal'
  
  -- Geographic operating zones
  active_zones text[],  -- 'casablanca', 'marrakech', 'rabat', etc.
  
  -- Stats
  total_missions integer NOT NULL DEFAULT 0,
  avg_rating decimal(3, 2),
  avg_response_time_hours decimal(5, 2),
  
  -- Honoraires baseline (peut etre override par mission)
  baseline_honoraire_mad decimal(10, 2),
  
  -- Status
  status text NOT NULL CHECK (status IN ('active', 'suspended', 'expired_agrement', 'inactive')) DEFAULT 'active',
  
  created_at timestamptz NOT NULL DEFAULT NOW(),
  updated_at timestamptz NOT NULL DEFAULT NOW()
);


-- insure_expert_assignments : Designations par carriers
CREATE TABLE insure_expert_assignments (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id uuid NOT NULL REFERENCES auth_tenants(id) ON DELETE CASCADE,
  
  carrier_tenant_id uuid NOT NULL REFERENCES auth_tenants(id),
  carrier_user_id uuid NOT NULL REFERENCES auth_users(id),  -- carrier_expert_manager qui designe
  
  expert_tenant_id uuid NOT NULL REFERENCES auth_tenants(id),
  expert_id uuid NOT NULL REFERENCES insure_experts(id),
  expert_user_id uuid NOT NULL REFERENCES auth_users(id),
  
  sinistre_id uuid NOT NULL,  -- FK insure_sinistres (Sprint 14)
  garage_tenant_id uuid REFERENCES auth_tenants(id),
  
  status text NOT NULL CHECK (status IN ('designated', 'accepted', 'rejected', 'in_progress', 'completed', 'cancelled')) DEFAULT 'designated',
  
  designated_at timestamptz NOT NULL DEFAULT NOW(),
  accepted_at timestamptz,
  rejected_at timestamptz,
  rejection_reason text,
  visit_scheduled_at timestamptz,
  visit_completed_at timestamptz,
  report_submitted_at timestamptz,
  completed_at timestamptz,
  
  -- Honoraire pour cette mission
  honoraire_mad decimal(10, 2),
  honoraire_invoice_id uuid,
  honoraire_payment_status text CHECK (honoraire_payment_status IN ('pending', 'paid', 'overdue')) DEFAULT 'pending',
  
  notes text,
  updated_at timestamptz NOT NULL DEFAULT NOW()
);


-- insure_expert_reports : Rapports expertise digitaux
CREATE TABLE insure_expert_reports (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id uuid NOT NULL REFERENCES auth_tenants(id) ON DELETE CASCADE,
  assignment_id uuid NOT NULL REFERENCES insure_expert_assignments(id) ON DELETE CASCADE,
  expert_id uuid NOT NULL REFERENCES insure_experts(id),
  
  -- Contenu rapport (JSONB structure)
  report_content jsonb NOT NULL,  -- structure : { sections: [...], findings: [...], recommendations: [...] }
  
  -- Photos
  photos_urls text[] NOT NULL DEFAULT '{}',
  
  -- Devis decision
  devis_id uuid NOT NULL,  -- FK repair_devis (Sprint 19/21)
  decision text NOT NULL CHECK (decision IN ('validated', 'modified', 'rejected')),
  decision_justification text NOT NULL,
  
  -- Modifications devis (si decision='modified')
  modifications jsonb,  -- structure : { lines_modified: [...], total_before: X, total_after: Y }
  
  -- Generated PDF
  pdf_url text,
  pdf_generated_at timestamptz,
  
  -- Signature electronique Barid eSign
  signature_id uuid REFERENCES docs_signatures(id),
  signed_at timestamptz,
  signature_legal_status text CHECK (signature_legal_status IN ('pending', 'signed', 'expired')) DEFAULT 'pending',
  
  -- Status
  status text NOT NULL CHECK (status IN ('draft', 'completed', 'signed', 'submitted_to_carrier')) DEFAULT 'draft',
  
  created_at timestamptz NOT NULL DEFAULT NOW(),
  updated_at timestamptz NOT NULL DEFAULT NOW()
);


-- insure_expert_signatures : Tracking signatures Barid eSign sur rapports experts
CREATE TABLE insure_expert_signatures (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id uuid NOT NULL REFERENCES auth_tenants(id) ON DELETE CASCADE,
  report_id uuid NOT NULL REFERENCES insure_expert_reports(id) ON DELETE CASCADE,
  expert_user_id uuid NOT NULL REFERENCES auth_users(id),
  
  barid_signature_id text NOT NULL,
  barid_signature_url text NOT NULL,
  barid_timestamp_anrt text NOT NULL,
  barid_legal_value text NOT NULL CHECK (barid_legal_value IN ('probante', 'simple')),
  
  signed_at timestamptz NOT NULL DEFAULT NOW(),
  
  -- Archivage legal 10 ans (loi 43-20)

### Pattern critique : modifyDevis

```typescript
async modifyDevis(
  reportId: string,
  modifications: DevisModifications
): Promise<void> {
  const report = await this.reportsRepo.findOneOrFail(reportId);
  const devis = await this.repairDevisRepo.findOneOrFail(report.devis_id);
  
  // Validation : modifications schema
  const validated = DevisModificationsSchema.parse(modifications);
  
  // Compute totals
  const totalBefore = new Decimal(devis.total_ttc);
  let totalAfter = totalBefore;
  
  const linesModifiedDetails: LineModificationDetail[] = [];
  
  for (const mod of validated.lines_modified) {
    const line = devis.lines.find(l => l.id === mod.line_id);
    if (!line) throw new BadRequestException(`Line ${mod.line_id} not found`);
    
    const oldAmount = new Decimal(line.total_ttc);
    const newAmount = new Decimal(mod.new_amount);
    
    totalAfter = totalAfter.minus(oldAmount).plus(newAmount);
    
    linesModifiedDetails.push({
      line_id: mod.line_id,
      description: line.description,
      old_quantity: line.quantity,
      new_quantity: mod.new_quantity ?? line.quantity,
      old_unit_price: line.unit_price,
      new_unit_price: mod.new_unit_price ?? line.unit_price,
      old_total_ttc: oldAmount.toFixed(2),
      new_total_ttc: newAmount.toFixed(2),
      reason: mod.reason,
    });
    
    // Update devis line
    await this.repairDevisLinesRepo.update(mod.line_id, {
      quantity: mod.new_quantity ?? line.quantity,
      unit_price: mod.new_unit_price ?? line.unit_price,
      total_ttc: newAmount.toFixed(2),
      modified_by_expert_at: new Date(),
      modified_by_expert_id: this.userContext.userId,
      modification_reason: mod.reason,
    });
  }
  
  // Update devis total
  await this.repairDevisRepo.update(devis.id, {
    total_ttc: totalAfter.toFixed(2),
    modified_by_expert_at: new Date(),
    status: 'quote_modified_by_expert',  // workflow transition
  });
  
  // Update report
  await this.reportsRepo.update(reportId, {
    decision: 'modified',
    decision_justification: validated.justification,
    modifications: {
      lines_modified: linesModifiedDetails,
      total_before: totalBefore.toFixed(2),
      total_after: totalAfter.toFixed(2),
    },
  });
  
  // Emit event
  await this.kafkaProducer.emit('insurtech.events.repair.devis.modified_by_expert', {
    reportId,
    devisId: devis.id,
    sinistreId: devis.sinistre_id,
    totalBefore: totalBefore.toFixed(2),
    totalAfter: totalAfter.toFixed(2),
    modifications: linesModifiedDetails,
  });
  
  // Audit log
  await this.auditLogService.log({
    entity_type: 'repair_devis',
    entity_id: devis.id,
    action: 'modified_by_expert',
    actor_user_id: this.userContext.userId,
    details: { totalBefore, totalAfter, modifications: linesModifiedDetails },
  });
  
  // Notify garage of modifications
  await this.notificationsService.notifyGarageDevisModifiedByExpert(devis.garage_tenant_id, devis.id);
}
```
