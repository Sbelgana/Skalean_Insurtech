 *   Tenant Tow N2         : tow_admin > tow_dispatcher > tow_driver

 *   - 00-pilotage/documentation/5-roles-permissions.md (matrice 26 roles x 130 permissions)

 *   - 00-pilotage/decisions/013-expert-acteur-central.md

/**
 * Type guard: is this role a Platform-level role (Niveau 1)?
 * Platform roles bypass tenant isolation (no tenant_id required in JWT).
 * MUST be checked before any tenant_id comparison logic.
 */
export function isPlatformRole(role: AuthRole): boolean {
  return role === AuthRole.SuperAdminPlatform || role === AuthRole.AnalystSupport;
}

/**
 * Type guard: is this role a Tenant-level role (Niveau 2)?
 * Tenant roles require tenant_id in JWT. Operations are scoped to that tenant.
 */
export function isTenantRole(role: AuthRole): boolean {
  return isBrokerRole(role) || isGarageRole(role) || isCarrierRole(role) || isExpertRole(role) || isTowRole(role);
}

/** Type guard: is this role specific to broker tenants (cabinet courtage)? */
export function isBrokerRole(role: AuthRole): boolean {
  return (
    role === AuthRole.BrokerAdmin ||
    role === AuthRole.BrokerUser ||
    role === AuthRole.BrokerAssistant
  );
}

/** Type guard: is this role specific to garage tenants (repair shop)? */
export function isGarageRole(role: AuthRole): boolean {
  return (
    role === AuthRole.GarageAdmin ||
    role === AuthRole.GarageChef ||
    role === AuthRole.GarageTechnicien ||
    role === AuthRole.GarageComptable ||
    role === AuthRole.GarageCommercial ||
    role === AuthRole.GaragePartsManager  // NOUVEAU v3.0
  );
}

/** Type guard: is this role specific to carrier tenants (insurance company)? -- NOUVEAU v3.0 */
export function isCarrierRole(role: AuthRole): boolean {
  return (
    role === AuthRole.CarrierAdmin ||
    role === AuthRole.CarrierClaimsManager ||
    role === AuthRole.CarrierFinance ||
    role === AuthRole.CarrierCompliance ||
    role === AuthRole.CarrierExpertManager ||
    role === AuthRole.CarrierPartnerManager
  );
}

/** Type guard: is this role specific to expert tenants (claims experts)? -- NOUVEAU v3.0 */
export function isExpertRole(role: AuthRole): boolean {
  return (
    role === AuthRole.ExpertIndependent ||
    role === AuthRole.ExpertFirmAdmin ||
    role === AuthRole.ExpertAssociate ||
    role === AuthRole.ExpertCarrierInternal
  );
}

/** Type guard: is this role specific to tow tenants (towing company)? -- NOUVEAU v3.0 */
export function isTowRole(role: AuthRole): boolean {
  return (
    role === AuthRole.TowAdmin ||
    role === AuthRole.TowDriver ||
    role === AuthRole.TowDispatcher
  );
}

/** Type guard: is this role the L3 Assure (end user in tenant)? */
export function isAssureRole(role: AuthRole): boolean {
  return role === AuthRole.Assure;
}

/** Type guard: is this role public (non-authenticated)? */
export function isPublicRole(role: AuthRole): boolean {
  return role === AuthRole.Prospect;
}

-- repo/packages/database/src/postgres/helpers/app-can-access-tenant.sql

CREATE OR REPLACE FUNCTION app_can_access_tenant(target_tenant uuid)
RETURNS boolean
LANGUAGE plpgsql
STABLE
SECURITY DEFINER
AS $$
BEGIN
  -- ============ COND 1 : SUPER ADMIN BYPASS ============
  IF current_setting('app.is_super_admin', true) = 'true' THEN
    RETURN true;
  END IF;
  
  -- ============ COND 2 : OWN TENANT ============
  IF current_setting('app.current_tenant_id', true) = target_tenant::text THEN
    RETURN true;
  END IF;
  
  -- ============ COND 3 : CROSS-TENANT AUTHORIZATION ============
  -- Check active authorization between current tenant and target tenant
  -- v3.0 : supports 7 types
  
  IF EXISTS (
    SELECT 1 
    FROM cross_tenant_authorizations cta
    WHERE 
      -- Active authorization
      cta.revoked_at IS NULL
      AND cta.expires_at > NOW()
      -- Match types v3.0 (7 valeurs)
      AND cta.type IN (
        -- Existing v2.2
        'broker_to_garage_assignment',
        'assure_to_garage_visit',
        'multi_tenant_user_access',
        -- New v3.0 (Sprint 7.5a)
        'client_to_tower_dispatch',
        'tower_to_garage_delivery',
        'garage_to_expert_request',
        'garage_to_carrier_quote'
      )
      -- Match current tenant <-> target tenant
      AND (
        (cta.from_tenant_id::text = current_setting('app.current_tenant_id', true) 
          AND cta.to_tenant_id = target_tenant)
        OR
        (cta.to_tenant_id::text = current_setting('app.current_tenant_id', true) 
          AND cta.from_tenant_id = target_tenant)
      )
  ) THEN
    RETURN true;
  END IF;
  
  -- Default : access denied
  RETURN false;
END;
$$;

-- Grant execute to applicative role
GRANT EXECUTE ON FUNCTION app_can_access_tenant(uuid) TO insurtech_app;

-- Documentation comment
COMMENT ON FUNCTION app_can_access_tenant(uuid) IS 
'Returns true if current session can access the target tenant.
Logic:
  1. Bypass if current user is super_admin (app.is_super_admin = true)
  2. Allow if target tenant = current tenant (app.current_tenant_id)
  3. Allow if active cross-tenant authorization exists (7 types v3.0)
  
v3.0 (Sprint 7.5a) : extended from 3 to 7 cross-tenant authorization types.
Reference : 00-pilotage/decisions/012-6-acteurs-ecosystem.md';

import { describe, it, expect } from 'vitest';
import {

    expect(garageRoles).toContain(AuthRole.GaragePartsManager);
  });

  it('should not allow new carrier roles to be platform roles', () => {
    expect(isPlatformRole(AuthRole.CarrierAdmin)).toBe(false);

        isGarageRole(role),
        isCarrierRole(role),
        isExpertRole(role),
        isTowRole(role),
        isAssureRole(role),
        isPublicRole(role)
      ];
      const trueCount = guards.filter(Boolean).length;
      expect(trueCount).toBe(1);
    }
  });
